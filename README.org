   (with-database ""
     (fn (db)
       (execute db "create table person (name, age);"
                mempty)
       (execute db "insert into person values (?, ?)"
                (make-list (Text "garlic0x1") (Int 26)))
       (execute db "insert into person values (?, ?)"
                (make-list (Text "foo") (Int -2)))
       (query db "select * from person" mempty))))
